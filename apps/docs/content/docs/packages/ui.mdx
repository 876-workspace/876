Shared shadcn/ui-style primitives and chart components.

## Import Pattern

```tsx
import { Button } from '@876/ui/button'
import { Input } from '@876/ui/input'
import { Dialog, DialogContent, DialogTrigger } from '@876/ui/dialog'
import { cn } from '@876/ui/lib/utils'
```

Import components directly from subpaths. The root export is for utilities and hooks.

## Example

```tsx
import { Button } from '@876/ui/button'
import { Input } from '@876/ui/input'

export function InviteForm() {
  return (
    <form className="grid gap-3">
      <Input type="email" placeholder="teammate@example.com" />
      <Button type="submit">Invite teammate</Button>
    </form>
  )
}
```

pnpm --filter @876/ui typecheck
